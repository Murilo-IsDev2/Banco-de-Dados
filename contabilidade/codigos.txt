SELECT 
    l.descricao,
    l.valor,
    l.data_lancamento,
    c.nome AS categorias
FROM
    lancamentos l
RIGHT JOIN
    categorias c 
    ON l.categoria_id = c.categoria_id;
WHERE 
    l.valor> 100;


SELECT 
    l.descricao,
    l.valor,
    l.data_lancamento,
    c.nome AS categorias,
    fp.nome AS forma_pagamento
FROM
    lancamentos l
JOIN
    categorias c 
    ON l.categoria_id = c.categoria_id
JOIN
    formas_pagamento fp
    ON l.forma_pagamento_id = fp.forma_pagamento_id
WHERE 
    l.valor> 100;


SELECT 
    valor,
    ROUND (valor * 0.08, 2) AS valor_do_imposto,
    ROUND (valor * 1.08, 2) AS valor_com_imposto,
    ROUND ((valor * 1.08 / 3), 2) AS valor_parcela_3x,
    mod(valor* 1.08, 10) AS quantas_parcelas_de_10
FROM 
    lancamentos;
